# Comando para inicializar o terraform
terraform init

# Comando para planejar o projeto
terraform plan

# Comando para aplicar o plano do terraform após inicialização ou alterações
terraform apply

# Comando para destruir todas as VMs criadas
terraform destroy
# OBS: Esse comando deleta todos os recursos definidos nos arquivos .tf do projeto atual.

# Comando para remover VM específica
terraform destroy -target=proxmox_virtual_environment_vm.vm_cloudinit

# Comando para remover mais de uma VM
terraform destroy -target=proxmox_virtual_environment_vm.vm_nginx

# IMPORTANTE: Para operações de deleção, sempre interessante rodar esse comando para informar o que será removido:
terraform plan -destroy

# Force Reset
terraform destroy -auto-approve
rm -rf .terraform terraform.tfstate*
terraform init
terraform apply

# Corrigir problema no template com IPS durante lançamento ansible
sudo truncate -s 0 /etc/machine-id
sudo rm /var/lib/dbus/machine-id
sudo ln -s /etc/machine-id /var/lib/dbus/machine-id

# Tirar a checagem
export ANSIBLE_HOST_KEY_CHECKING=False

# Teste de comando PING
ansible all -i hosts.ini -m ping

# Parar o dpkg após primeira iniciação de clonagem através do ansible
ansible nginx -i hosts.ini -m command -a "dpkg --configure -a" -b

# Iniciar o Playbook
ansible-playbook -i hosts.ini playbook-web.yml
